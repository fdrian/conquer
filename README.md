$ cd conquer/

$ bash conquer.sh --help
usage: bash conquer.sh 
usage: bash conquer.sh <COUNTRY-CODE>

options:
  -h, --help            show this help message and exit  
  -cc <COUNTRY-CODE>,   Country Code to Connect
